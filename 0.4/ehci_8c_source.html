178 static tusb_error_t hcd_controller_init(uint8_t hostid)
179 {
180  ehci_registers_t* const regs = get_operational_register(hostid);
181 
182  //------------- CTRLDSSEGMENT Register (skip) -------------//
183  //------------- USB INT Register -------------//
184  regs->usb_int_enable = 0; // 1. disable all the interrupt
185 #ifndef _TEST_ // the fake controller does not have write-to-clear behavior
186  regs->usb_sts = EHCI_INT_MASK_ALL; // 2. clear all status
187 #endif
188  regs->usb_int_enable = EHCI_INT_MASK_ERROR | EHCI_INT_MASK_PORT_CHANGE |
189 #if EHCI_PERIODIC_LIST
190  EHCI_INT_MASK_NXP_PERIODIC |
191 #endif
192  EHCI_INT_MASK_ASYNC_ADVANCE | EHCI_INT_MASK_NXP_ASYNC;
193 
194  //------------- Asynchronous List -------------//
195  ehci_qhd_t * const async_head = get_async_head(hostid);
196  memclr_(async_head, sizeof(ehci_qhd_t));
197 
198  async_head->next.address = (uint32_t) async_head; // circular list, next is itself
199  async_head->next.type = EHCI_QUEUE_ELEMENT_QHD;
200  async_head->head_list_flag = 1;
201  async_head->qtd_overlay.halted = 1; // inactive most of time
202  async_head->qtd_overlay.next.terminate = 1; // TODO removed if verified
203 
204  regs->async_list_base = (uint32_t) async_head;
205 
206 #if EHCI_PERIODIC_LIST
207  //------------- Periodic List -------------//
208  // Build the polling interval tree with 1 ms, 2 ms, 4 ms and 8 ms (framesize) only
209 
210  for(uint32_t i=0; i<4; i++)
211  {
212  ehci_data.period_head_arr[ hostid_to_data_idx(hostid) ][i].interrupt_smask = 1; // queue head in period list must have smask non-zero
213  ehci_data.period_head_arr[ hostid_to_data_idx(hostid) ][i].qtd_overlay.halted = 1; // dummy node, always inactive
214  }
215 
216  ehci_link_t * const framelist = get_period_frame_list(hostid);
217  ehci_link_t * const period_1ms = get_period_head(hostid, 1);
218  // all links --> period_head_arr[0] (1ms)
219  // 0, 2, 4, 6 etc --> period_head_arr[1] (2ms)
220  // 1, 5 --> period_head_arr[2] (4ms)
221  // 3 --> period_head_arr[3] (8ms)
222 
223  // TODO EHCI_FRAMELIST_SIZE with other size than 8
224  for(uint32_t i=0; i<EHCI_FRAMELIST_SIZE; i++)
225  {
226  framelist[i].address = (uint32_t) period_1ms;
227  framelist[i].type = EHCI_QUEUE_ELEMENT_QHD;
228  }
229 
230  for(uint32_t i=0; i<EHCI_FRAMELIST_SIZE; i+=2)
231  {
232  list_insert(framelist + i, get_period_head(hostid, 2), EHCI_QUEUE_ELEMENT_QHD);
233  }
234 
235  for(uint32_t i=1; i<EHCI_FRAMELIST_SIZE; i+=4)
236  {
237  list_insert(framelist + i, get_period_head(hostid, 4), EHCI_QUEUE_ELEMENT_QHD);
238  }
239 
240  list_insert(framelist+3, get_period_head(hostid, 8), EHCI_QUEUE_ELEMENT_QHD);
241 
242  period_1ms->terminate = 1;
243 
244  regs->periodic_list_base = (uint32_t) framelist;
245 #else
246  regs->periodic_list_base = 0;
247 #endif
248 
249  //------------- TT Control (NXP only) -------------//
250  regs->tt_control = 0;
251 
252  //------------- USB CMD Register -------------//
253 
254  regs->usb_cmd |= BIT_(EHCI_USBCMD_POS_RUN_STOP) | BIT_(EHCI_USBCMD_POS_ASYNC_ENABLE)
255 #if EHCI_PERIODIC_LIST
256  | BIT_(EHCI_USBCMD_POS_PERIOD_ENABLE) // TODO enable period list only there is int/iso endpoint
257 #endif
258  | ((EHCI_CFG_FRAMELIST_SIZE_BITS & BIN8(011)) << EHCI_USBCMD_POS_FRAMELIST_SZIE)
259  | ((EHCI_CFG_FRAMELIST_SIZE_BITS >> 2) << EHCI_USBCMD_POS_NXP_FRAMELIST_SIZE_MSB);
260 
261  //------------- ConfigFlag Register (skip) -------------//
262 
263  regs->portsc_bit.port_power = 1; // enable port power
264 
265  return TUSB_ERROR_NONE;
266 }

269 {
270  ehci_registers_t* const regs = get_operational_register(hostid);
271  timeout_timer_t timeout;
272 
273  regs->usb_cmd_bit.run_stop = 0;
274

300 tusb_error_t hcd_pipe_control_xfer(uint8_t dev_addr, tusb_control_request_t const * p_request, uint8_t data[])
301 {
302  ehci_qhd_t * const p_qhd = get_control_qhd(dev_addr);
303 
304  ehci_qtd_t *p_setup = get_control_qtds(dev_addr);
305  ehci_qtd_t *p_data = p_setup + 1;
306  ehci_qtd_t *p_status = p_setup + 2;
307 
308  //------------- SETUP Phase -------------//
309  qtd_init(p_setup, (uint32_t) p_request, 8);
310  p_setup->pid = EHCI_PID_SETUP;
311  p_setup->next.address = (uint32_t) p_data;
312 
313  //------------- DATA Phase -------------//
314  if (p_request->wLength > 0)
315  {
316  qtd_init(p_data, (uint32_t) data, p_request->wLength);
317  p_data->data_toggle = 1;
318  p_data->pid = p_request->bmRequestType_bit.direction ? EHCI_PID_IN : EHCI_PID_OUT;
319  }else
320  {
321  p_data = p_setup;
322  }
323  p_data->next.address = (uint32_t) p_status;
324 
325  //------------- STATUS Phase -------------//
326  qtd_init(p_status, 0, 0); // zero-length data
327  p_status->int_on_complete = 1;
328  p_status->data_toggle = 1;
329  p_status->pid = p_request->bmRequestType_bit.direction ? EHCI_PID_OUT : EHCI_PID_IN; // reverse direction of data phase
330  p_status->next.terminate = 1;
331 
332  //------------- Attach TDs list to Control Endpoint -------------//
333  p_qhd->p_qtd_list_head = p_setup;
334  p_qhd->p_qtd_list_tail = p_status;
335 
336  p_qhd->qtd_overlay.next.address = (uint32_t) p_setup;
337 
338  return TUSB_ERROR_NONE;
339 }

500 // async_advance is handshake between sw stack & ehci controller where ehci free all memory from an deleted queue head.
501 // In tinyusb, queue head is only removed when device is unplugged. So only control queue head is checked if removing
502 static void async_advance_isr(ehci_qhd_t * const async_head)
503 {
504  // TODO do we need to close addr0
505  if (async_head->is_removing) // closing control pipe of addr0
506  {
507  async_head->is_removing = 0;
508  async_head->p_qtd_list_head = async_head->p_qtd_list_tail = NULL;
509  async_head->qtd_overlay.halted = 1;
510 
511  usbh_devices[0].state = TUSB_DEVICE_STATE_UNPLUG;
512  }
513 
514  for(uint8_t relative_dev_addr=0; relative_dev_addr < TUSB_CFG_HOST_DEVICE_MAX; relative_dev_addr++)
515  {

556 static void qhd_xfer_complete_isr(ehci_qhd_t * p_qhd)
557 {
558  uint8_t max_loop = 0;
559  tusb_xfer_type_t const xfer_type = qhd_get_xfer_type(p_qhd);
560 
561  // free all TDs from the head td to the first active TD
562  while(p_qhd->p_qtd_list_head != NULL && !p_qhd->p_qtd_list_head->active
563  && max_loop < EHCI_MAX_QTD)
564  {
565  // TD need to be freed and removed from qhd, before invoking callback
566  bool is_ioc = (p_qhd->p_qtd_list_head->int_on_complete != 0);
567  p_qhd->total_xferred_bytes += p_qhd->p_qtd_list_head->expected_bytes - p_qhd->p_qtd_list_head->total_bytes;
568 
569  p_qhd->p_qtd_list_head->used = 0; // free QTD
570  qtd_remove_1st_from_qhd(p_qhd);
571 
572  if (is_ioc) // end of request
573  { // call USBH callback
574  usbh_xfer_isr( qhd_create_pipe_handle(p_qhd, xfer_type),
575  p_qhd->class_code, TUSB_EVENT_XFER_COMPLETE,
576  p_qhd->total_xferred_bytes - (xfer_type == TUSB_XFER_CONTROL ? 8 : 0) ); // subtract setup packet size if control,
577  p_qhd->total_xferred_bytes = 0;
578  }
579 
580  max_loop++;
581  }
582 }

637 static void qhd_xfer_error_isr(ehci_qhd_t * p_qhd)
638 {
639  if ( (p_qhd->device_address != 0 && p_qhd->qtd_overlay.halted) || // addr0 cannot be protocol STALL
640  qhd_has_xact_error(p_qhd) )
641  { // current qhd has error in transaction
642  tusb_xfer_type_t const xfer_type = qhd_get_xfer_type(p_qhd);
643  tusb_event_t error_event;
644 
645  // no error bits are set, endpoint is halted due to STALL
646  error_event = qhd_has_xact_error(p_qhd) ? TUSB_EVENT_XFER_ERROR : TUSB_EVENT_XFER_STALLED;
647 
648  p_qhd->total_xferred_bytes += p_qhd->p_qtd_list_head->expected_bytes - p_qhd->p_qtd_list_head->total_bytes;
649 
650  // TODO skip unplugged device
651  if ( TUSB_EVENT_XFER_ERROR == error_event ) hal_debugger_breakpoint();
652 
653  p_qhd->p_qtd_list_head->used = 0; // free QTD
654  qtd_remove_1st_from_qhd(p_qhd);
655 
656  if ( TUSB_XFER_CONTROL == xfer_type )
657  {
658  p_qhd->total_xferred_bytes -= min8_of(8, p_qhd->total_xferred_bytes); // subtract setup size
659 
660  // control cannot be halted --> clear all qtd list
661  p_qhd->p_qtd_list_head = NULL;
662  p_qhd->p_qtd_list_tail = NULL;
663 
664  p_qhd->qtd_overlay.next.terminate = 1;
665  p_qhd->qtd_overlay.alternate.terminate = 1;
666  p_qhd->qtd_overlay.halted = 0;
667 
668  ehci_qtd_t *p_setup = get_control_qtds(p_qhd->device_address);
669  ehci_qtd_t *p_data = p_setup + 1;
670  ehci_qtd_t *p_status = p_setup + 2;
671 
672  p_setup->used = p_data->used = p_status->used = 0;
673  }
674 
675  // call USBH callback
676  usbh_xfer_isr( qhd_create_pipe_handle(p_qhd, xfer_type),
677  p_qhd->class_code, error_event,
678  p_qhd->total_xferred_bytes);
679 
680  p_qhd->total_xferred_bytes = 0;
681  }
682 }

733 //------------- Host Controller Driver's Interrupt Handler -------------//
734 void hcd_isr(uint8_t hostid)
735 {
736  ehci_registers_t* const regs = get_operational_register(hostid);
737 
738  uint32_t int_status = regs->usb_sts;
739  int_status &= regs->usb_int_enable;
740 
741  regs->usb_sts |= int_status; // Acknowledge handled interrupt
742 
743  if (int_status == 0) return;
744 
745  if (int_status & EHCI_INT_MASK_PORT_CHANGE)
746  {
747  uint32_t port_status = regs->portsc & EHCI_PORTSC_MASK_ALL;
748 
749  if (regs->portsc_bit.connect_status_change)
750  {
751  port_connect_status_change_isr(hostid);
752  }
753 
754  regs->portsc |= port_status; // Acknowledge change bits in portsc
755  }
756 
757  if (int_status & EHCI_INT_MASK_ERROR)
758  {
759  xfer_error_isr(hostid);
760  }
761 
762  //------------- some QTD/SITD/ITD with IOC set is completed -------------//
763  if (int_status & EHCI_INT_MASK_NXP_ASYNC)
764  {
765  async_list_xfer_complete_isr( get_async_head(hostid) );
766  }
767 
768 #if EHCI_PERIODIC_LIST // TODO refractor/group this together
769  if (int_status & EHCI_INT_MASK_NXP_PERIODIC)
770  {
771  for (uint8_t i=1; i <= EHCI_FRAMELIST_SIZE; i *= 2)
772  {
773  period_list_xfer_complete_isr( hostid, i );
774  }
775  }
776 #endif
777 
778  //------------- There is some removed async previously -------------//
779  if (int_status & EHCI_INT_MASK_ASYNC_ADVANCE) // need to place after EHCI_INT_MASK_NXP_ASYNC
780  {
781  async_advance_isr( get_async_head(hostid) );
782  }
783 }

938 static void qhd_init(ehci_qhd_t *p_qhd, uint8_t dev_addr, uint16_t max_packet_size, uint8_t endpoint_addr, uint8_t xfer_type, uint8_t interval)
939 {
940  // address 0 is used as async head, which always on the list --> cannot be cleared (ehci halted otherwise)
941  if (dev_addr != 0)
942  {
943  memclr_(p_qhd, sizeof(ehci_qhd_t));
944  }
945 
946  p_qhd->device_address = dev_addr;
947  p_qhd->non_hs_period_inactive_next_xact = 0;
948  p_qhd->endpoint_number = endpoint_addr & 0x0F;
949  p_qhd->endpoint_speed = usbh_devices[dev_addr].speed;
950  p_qhd->data_toggle_control = (xfer_type == TUSB_XFER_CONTROL) ? 1 : 0;
951  p_qhd->head_list_flag = (dev_addr == 0) ? 1 : 0; // addr0's endpoint is the static asyn list head
952  p_qhd->max_package_size = max_packet_size;
953  p_qhd->non_hs_control_endpoint = ((TUSB_XFER_CONTROL == xfer_type) && (p_qhd->endpoint_speed != TUSB_SPEED_HIGH)) ? 1 : 0;
954  p_qhd->nak_count_reload = 0;
955 
956  // Bulk/Control -> smask = cmask = 0
957  // TODO Isochronous
958  if (TUSB_XFER_INTERRUPT == xfer_type)
959  {
960  if (TUSB_SPEED_HIGH == p_qhd->endpoint_speed)
961  {
962  ASSERT_INT_WITHIN(1, 16, interval, VOID_RETURN);
963  if ( interval < 4) // sub milisecond interval
964  {
965  p_qhd->interval_ms = 0;
966  p_qhd->interrupt_smask = (interval == 1) ? BIN8(11111111) :
967  (interval == 2) ? BIN8(10101010) : BIN8(01000100);
968  }else
969  {
970  p_qhd->interval_ms = (uint8_t) min16_of( 1 << (interval-4), 255 );
971  p_qhd->interrupt_smask = BIT_(interval % 8);
972  }
973  }else
974  {
975  ASSERT( 0 != interval, VOID_RETURN);
976  // Full/Low: 4.12.2.1 (EHCI) case 1 schedule start split at 1 us & complete split at 2,3,4 uframes
977  p_qhd->interrupt_smask = 0x01;
978  p_qhd->non_hs_interrupt_cmask = BIN8(11100);
979  p_qhd->interval_ms = interval;
980  }
981  }else
982  {
983  p_qhd->interrupt_smask = p_qhd->non_hs_interrupt_cmask = 0;
984  }
985 
986  p_qhd->hub_address = usbh_devices[dev_addr].hub_addr;
987  p_qhd->hub_port = usbh_devices[dev_addr].hub_port;
988  p_qhd->mult = 1; // TODO not use high bandwidth/park mode yet
989 
990  //------------- HCD Management Data -------------//
991  p_qhd->used = 1;
992  p_qhd->is_removing = 0;
993  p_qhd->p_qtd_list_head = NULL;
994  p_qhd->p_qtd_list_tail = NULL;
995  p_qhd->pid_non_control = (endpoint_addr & 0x80) ? EHCI_PID_IN : EHCI_PID_OUT; // PID for TD under this endpoint
996 
997  //------------- active, but no TD list -------------//
998  p_qhd->qtd_overlay.halted = 0;
999  p_qhd->qtd_overlay.next.terminate = 1;
1000  p_qhd->qtd_overlay.alternate.terminate = 1;
1001  if (TUSB_XFER_BULK == xfer_type && p_qhd->endpoint_speed == TUSB_SPEED_HIGH && p_qhd->pid_non_control == EHCI_PID_OUT)
1002  {
1003  p_qhd->qtd_overlay.pingstate_err = 1; // do PING for Highspeed Bulk OUT, EHCI section 4.11
1004  }
1005 }

1040 static ehci_link_t* list_find_previous_item(ehci_link_t* p_head, ehci_link_t* p_current)
