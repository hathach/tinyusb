196 /******************************************************************************
197  *
198  * Description:
199  * When the SST39VF160x/320x are in the internal Program operation, any
200  * attempt to read DQ7 will produce the complement of the true data. Once
201  * the Program operation is completed, DQ7 will produce true data. Note
202  * that even though DQ7 may have valid data immediately following the
203  * completion of an internal Write operation, the remaining data outputs
204  * may still be invalid: valid data on the entire data bus will appear in
205  * subsequent successive Read cycles after an interval of 1 �s. During
206  * internal Erase operation, any attempt to read DQ7 will produce a '0'.
207  * Once the internal Erase operation is completed, DQ7 will produce a '1'.
208  *
209  * Parameters:
210  * addr The device address
211  * data The original (true) data
212  * timeout Maximum number of loops to delay
213  *
214  * Returns:
215  * TRUE if success
216  *
217  *****************************************************************************/
218 static uint16_t check_data_polling(uint32_t addr, uint16_t data, uint32_t timeout)
219 {
220  volatile uint16_t *p = (uint16_t*) addr;
221  uint16_t true_data = data & 0x80;
222  int i;
223 
224  for (i = 0; i < timeout; i++)
225  {
226  if ( true_data == (*p &0x80) )
227  {
228  TIM_Waitus(1);
229  return (TRUE);
230  }
231  }
232  return (FALSE);
233 }
234 
235 /******************************************************************************
236  *
237  * Description:
238  * During the internal Program or Erase operation, any consecutive attempts
239  * to read DQ6 will produce alternating �1�s and �0�s, i.e., toggling
240  * between 1 and 0. When the internal Program or Erase operation is
241  * completed, the DQ6 bit will stop toggling. The device is then ready
242  * for the next operation.
243  *
244  * Parameters:
245  * addr The device address
246  * timeout Maximum number of loops to delay
247  *
248  * Returns:
249  * TRUE if success
250  *
251  *****************************************************************************/
252 static uint16_t check_toggle_ready(uint32_t addr, uint32_t timeout)
253 {
254  volatile uint16_t *p = (uint16_t*) addr;
255  uint16_t predata, currdata;
256  int i;
257 
258  predata = *p & 0x40;
259  for (i = 0; i < timeout; i++)
260  {
261  currdata = *p & 0x40;
262  if (predata == currdata)
263  {
264  TIM_Waitus(1);
265  return (TRUE);
266  }
267  predata = currdata;
268  }
269  return (FALSE);
270 }
